      class node
      {
      public:
              int data;
              node * next;
          node(int val)
          {
              data = val;
              next = NULL;
          }
      };

      void insert(node * &head, int val)
      {
          node * n = new node(val);
          if(head == NULL)
          {
              head = n;
              return;
          }
          node *p = head;
          while(p->next != NULL)
          {
              p = p->next;
          }
          p->next = n;
      }

      node* reverse(node *&head)
      {
          node *current = head;
          node *nextp;
          node *prev = NULL;
          while(current != NULL)
          {
              nextp = current->next;
              current->next = prev;
              prev = current;
              current = nextp;
          }
      return prev;
      }

      node* reverse_rec(node* &head)
      {
          if(head==NULL || head->next == NULL)
          {
              return head;
          }
          node* newhead = reverse_rec(head->next);
          head->next->next = head;
          head->next = NULL;

          return newhead;
      }
